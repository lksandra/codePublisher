[-]
    {

      name = name.substr(6, name.size() - 6);
      if(predicate)

      else
        return failed(name);
    }

  protected:

[-]
    {
      std::cout << "\n  passed: " << name;
      return true;
    }

[-]

      std::cout << "\n  failed: " << name;
      return false;
